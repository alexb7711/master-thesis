\documentclass[float=false, crop=true]{standalone}

\usepackage
{
    pgfplots,                                                                   % Plot in LaTeX
    pgfplotstable,
    xfp,
}
\usetikzlibrary{automata, positioning, arrows.meta}                             % Tikz macros

\begin{document}

\pgfplotstableread[col sep= comma]{data/sa-nonlinear-schedule.csv}\scheduledatanonlinear

\begin{tikzpicture}
\pgfplotstablegetcolsof{\scheduledatanonlinear}
\pgfmathtruncatemacro\NumCols{\pgfplotsretval/3 - 1}
\begin{axis}[title={SA Non-Linear Schedule},
      scale only axis,
      height=0.325\paperheight, width=0.75\paperwidth,
      grid=both,
      ytick distance=5,
      minor y tick num= 4,
      xlabel={Time [hr]}, ylabel={Charging Queue},
      xmin=4, xmax=23.99]
    \foreach \i in {0, ..., \NumCols}
    {
        \pgfmathsetmacro{\v}{\fpeval{3*\i} + 0}
        \pgfmathsetmacro{\u}{\fpeval{3*\i} + 1}
        \pgfmathsetmacro{\p}{\fpeval{3*\i} + 2}
        \addplot +[green!60!black, only marks, mark=otimes, error bars/.cd, error bar style={solid}, x dir=plus, x explicit]
                  table[x index=\u, x error index=\p, y index =\v, unbounded coords=discard]{\scheduledatanonlinear};
    }
\end{axis}
\end{tikzpicture}

\end{document}
