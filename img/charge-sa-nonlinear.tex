\documentclass[float=false, crop=true]{standalone}

\usepackage
{
    tikz,                                                                       % Generate figures in LaTeX
    pgfplots,                                                                   % Plot in LaTeX
    pgfplotstable,
    xfp,
}
\pgfplotsset{compat=1.18}
\usetikzlibrary{automata, positioning, arrows.meta}                             % Tikz macros

\begin{document}

\pgfplotstableread[col sep=comma]{data/sa-nonlinear-charge.csv}\chargedata

\begin{tikzpicture}
\pgfplotstablegetcolsof{\chargedata}
\pgfmathtruncatemacro\NumCols{\pgfplotsretval/2-1}
\begin{axis}[title={SA Exhaustive Charges},
      scale only axis,
      height=0.325\paperheight, width=0.75\paperwidth,
      grid=both,
      xlabel={Time [hr]},
      xmin=0, xmax=24,
      ylabel={Charge [kWh]}]
    \foreach \i in {0, ..., \NumCols}
    {
        \pgfmathsetmacro{\u}{\fpeval{2*\i} + 0}
        \pgfmathsetmacro{\etas}{\fpeval{2*\i} + 1}
        \addplot +[mark=none] table[x index=\u, y index =\etas, unbounded coords=jump]{\chargedata};
    }
\end{axis}
\end{tikzpicture}

\end{document}
